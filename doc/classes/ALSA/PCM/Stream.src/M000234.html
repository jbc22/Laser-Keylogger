    def change_software_parameters
      sw_params = software_parameters

      begin
        yield sw_params

        ALSA::try_to "set sw parameters" do
          ALSA::PCM::Native::sw_params self.handle, sw_params.handle
        end
      ensure
        sw_params.free
      end
    end
