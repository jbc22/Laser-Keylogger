    def avail_min
      value = nil
      ALSA::try_to "get period time" do
        value_pointer = FFI::MemoryPointer.new(:int)
        error_code = ALSA::PCM::Native::sw_params_get_avail_min self.handle, value_pointer
        value = value_pointer.read_int
        value_pointer.free
        error_code
      end
      value
    end
